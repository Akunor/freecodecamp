def get_bingo_letter(n):
    bingo_letter = ''

    if n <= 15:
        bingo_letter = 'B'
    elif n <= 30:
        bingo_letter = 'I'
    elif n <= 45:
        bingo_letter = 'N'
    elif n <= 60:
        bingo_letter = 'G'
    elif n <= 75:
        bingo_letter = 'O'

    return bingo_letter
